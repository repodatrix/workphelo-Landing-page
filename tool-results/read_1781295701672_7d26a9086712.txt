   200→   200→/** Animated count-up number */
   201→   201→function CountUp({ target, duration = 2, suffix = '' }: { target: number; duration?: number; suffix?: string }) {
   202→   202→  const [count, setCount] = useState(0);
   203→   203→  const ref = useRef<HTMLSpanElement>(null);
   204→   204→  const hasAnimated = useRef(false);
   205→   205→
   206→   206→  useEffect(() => {
   207→   207→    const observer = new IntersectionObserver(
   208→   208→      ([entry]) => {
   209→   209→        if (entry.isIntersecting && !hasAnimated.current) {
   210→   210→          hasAnimated.current = true;
   211→   211→          const start = performance.now();
   212→   212→          const step = (now: number) => {
   213→   213→            const progress = Math.min((now - start) / (duration * 1000), 1);
   214→   214→            const eased = 1 - Math.pow(1 - progress, 3);
   215→   215→            setCount(Math.round(eased * target));
   216→   216→            if (progress < 1) requestAnimationFrame(step);
   217→   217→          };
   218→   218→          requestAnimationFrame(step);
   219→   219→        }
   220→   220→      },
   221→   221→      { threshold: 0.5 }
   222→   222→    );
   223→   223→    if (ref.current) observer.observe(ref.current);
   224→   224→    return () => observer.disconnect();
   225→   225→  }, [target, duration]);
   226→   226→
   227→   227→  return <span ref={ref}>{count}{suffix}</span>;
   228→   228→}
